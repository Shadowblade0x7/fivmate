    <script>
        // --- 1. Custom Cursor Logic ---
        const cursor = document.getElementById('cursor');
        document.addEventListener('mousemove', (e) => {
            cursor.style.left = e.clientX + 'px';
            cursor.style.top = e.clientY + 'px';
            
            // Create Sparkle Trail
            const trail = document.createElement('div');
            trail.classList.add('trail');
            trail.style.left = e.clientX + 'px';
            trail.style.top = e.clientY + 'px';
            // Randomize sparkle size
            const size = Math.random() * 5;
            trail.style.width = size + 'px';
            trail.style.height = size + 'px';
            document.body.appendChild(trail);

            setTimeout(() => trail.remove(), 1000);
        });

        // --- 2. The Easter Egg (Theme Toggle) ---
        let isNight = true;
        const body = document.body;
        const lantern = document.getElementById('lantern');

        function toggleTheme() {
            if (isNight) {
                body.classList.remove('night');
                body.classList.add('day');
                lantern.classList.add('lit');
            } else {
                body.classList.remove('day');
                body.classList.add('night');
                lantern.classList.remove('lit');
            }
            isNight = !isNight;
        }

        // --- 3. Countdown Logic (Fixed Date) ---
        const targetDate = new Date("April 10, 2026 00:00:00").getTime();

        setInterval(() => {
            const now = new Date().getTime();
            const gap = targetDate - now;

            if (gap < 0) {
                document.querySelector('.countdown').innerHTML = "<div>The Blessed Union</div>";
                return;
            }

            const second = 1000;
            const minute = second * 60;
            const hour = minute * 60;
            const day = hour * 24;

            document.getElementById("d").innerText = Math.floor(gap / day);
            document.getElementById("h").innerText = Math.floor((gap % day) / hour);
            document.getElementById("m").innerText = Math.floor((gap % hour) / minute);
            document.getElementById("s").innerText = Math.floor((gap % minute) / second);
        }, 1000);
    </script>
